#include <stdio.h>

int main() {
  int M, N;
  int temp;
  scanf("%d %d", &M, &N);
  int GCD = 0;
  
  if(M>N) {
    M = M;
    N = N;
  }
  else {
    temp = M;
    M = N;
    N = temp;
  }
  
  // int arr[N];
  
  for(int i=1;i<=N;i++) {
    if((M%i==0)&&(N%i==0)) {
      GCD = i;
    }
  }
  
  printf("%d\n", GCD);
  printf("%d", M*N/GCD);
  return 0;
}
