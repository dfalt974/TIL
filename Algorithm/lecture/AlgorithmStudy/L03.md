```c
#include <stdio.h>

int main() {
	int n;
	int car[5];
	int cnt = 0;
	
	scanf("%d", &n);
	for (int i = 0; i < 5; i++) {
		scanf("%d", &car[i]);
	}

	for (int i = 0; i < 5; i++) {
		if (car[i] == n) {
			cnt += 1;
		}
	}

	printf("%d", cnt);

	return 0;
}
```
